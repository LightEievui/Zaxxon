66void Background::update(sf::RenderWindow& window, sf::View& mainView,
67 float gameSpeed, sf::Texture* spritesheet,
68 std::vector<Obstacle*>& obstacles,
69 std::vector<Enemy*>& enemies, Player& player,
70 std::vector<Wall*>& walls,
71 std::vector<ZapWall*>& zapWalls
72)
73{
74 //checks the transition between stages
75 if (backgroundFinished(mainView))
76 {
77 switch (stage)
78 {
79 case INITIAL:
80 stage = SPACE;
81 back.setTexture(space);
82 resetPos(mainView, player, 0);
83 break;
84 case SPACE:
85 stage = BOSS;
86 back.setTexture(boss);
87 resetPos(mainView, player, 0);
88 break;
89 case BOSS:
90 stage = BOSSFIGHT;
91 }
92
93 generateObstacles(stage, obstacles, spritesheet, walls, zapWalls);
94 generateWaves(stage, enemies, spritesheet,
95 static_cast<int>(player.getPos().z));
96 }
97
98 if (!backgroundFinished(mainView))
99 {
100 float modifier = 1;
101 if (stage == SPACE) // space moves 2/3 speed
102 modifier = 0.66f;
103 mainView.move(
104 translateTo2d(sf::Vector3f(0, 0, -1.3f * gameSpeed * modifier)));
105 }
106
107 // spawn waves that have gone past the z set in queue
108 if (!waveQueue.empty() && player.getPos().z < waveQueue.front().first)
109 {
110 Enemy::spawnWave(enemies, spritesheet,
111 static_cast<int>(player.getPos().z),
112 waveQueue.front().second);
113 waveQueue.pop();
114 }
115
116 window.draw(back);
117}
